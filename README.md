# usage
```sh
$ go run ./cmd/envoy-xds-server/main.go --configFile ./config.yaml
$ go run ./cmd/envoy-xds-server/main.go --configFile ./1000.yaml
$ go run ./cmd/envoy-xds-server/main.go --configFile ./2000.yaml
$ go run ./cmd/envoy-xds-server/main.go --configFile ./2100.yaml
```

DROP DATABASE IF EXISTS `fedx-2100`;
CREATE DATABASE IF NOT EXISTS `fedx-2100`;

USE `fedx-1000`;
USE `fedx-2000`;
USE `fedx-2100`;
DROP TABLE IF EXISTS `peer`;
CREATE TABLE IF NOT EXISTS `peer`(
  `id` VARCHAR(255) NOT NULL,
  `mode` ENUM('tcp','http') NOT NULL DEFAULT 'tcp',
  `host` VARCHAR(255) NOT NULL DEFAULT '0.0.0.0',
  `port` INT NOT NULL DEFAULT 10000,
  `tls_root_ca` TEXT,
  `tls_private_key` TEXT,
  `tls_private_ca` TEXT,
  `is_deleted` BOOLEAN NOT NULL DEFAULT FALSE
);

# Setup
```sh
$ go mod init envoy-xds-server
$ go get -u github.com/go-sql-driver/mysql
$ go mod tidy

# for hello world
$ go run ./cmd/hello-world/main.go
$ go build -o ./bin/hello-world ./cmd/hello-world/main.go

# for envoy-xds-server
$ export GOPATH="$PWD"
$ go run ./cmd/envoy-xds-server/main.go
$ go build -o ./bin/envoy-xds-server ./cmd/envoy-xds-server/main.go
$ go build -o ./bin/envoy-xds-server ./internal/example/main/main.go
```
